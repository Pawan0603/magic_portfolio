## Step 2: Create a Workflow

Inside n8nâ€™s UI:  
1. Create a new workflow

## Step 3: Expose Webhook in Next.js

From your Next.js app, you can send requests to the n8n webhook whenever an action happens (like form submission).

<CodeBlock className="my-24"
    compact
    codeInstances={[
        {
            code: `// pages/api/submit.ts
import type { NextApiRequest, NextApiResponse } from "next";

export default async function handler(req: NextApiRequest, res: NextApiResponse) {
  if (req.method === "POST") {
    const response = await fetch("http://localhost:5678/webhook/12345", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(req.body),
    });

    const data = await response.json();
    return res.status(200).json({ success: true, data });
  }

  res.status(405).json({ error: "Method Not Allowed" });
}`,
            label: "API Route",
            language: "ts"
        },
    ]}
    copyButton
/>

Now whenever you send a POST request to `/api/submit` in Next.js, it will trigger your n8n workflow.
